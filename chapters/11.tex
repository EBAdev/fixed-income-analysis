\chapter{Stochastic intensity reduced form model}
Letting the intensity be stochastic results in the default process $N_{t}$ being a Cox process. In this setup the survival probabilities are given by

$$
\mathbb{Q}_{t}[\tau>T]=\mathbb{E}_{t}\left[\mathrm{e}^{-\int_{t}^{T} \lambda_{s} \mathrm{~d} s}\right]
$$

This is completely analogue to pricing of risk-free ZCBs, so all the short interest rate modeling machinery can be used.

CIR intensity: Letting the default intensity follow a CIR process

$$
\mathrm{d} \lambda_{t}=k(\bar{\lambda}-\lambda) \mathrm{d} t+\sigma \sqrt{\lambda_{t}} \mathrm{~d} W_{t},
$$

we have a tractable specification. The intensity varies ramdomly with a tendency to revert to its long-run mean $\bar{\lambda}$ ( $k$ determines the speed of reversion). With the Feller conditions

$$
2 k \bar{\lambda}>\sigma^{2}, \quad \text { and } \quad \lambda_{0}>0,
$$

it can be shown that $\lambda_{t}$ is strictly positive for all $t$. In this setting the survival probability is

$$
\mathbb{Q}_{t}[\tau>T]=\mathrm{e}^{a(T-t)-b(T-t) \lambda_{t}}
$$

where $a$ and $b$ are deterministic functions of time to maturity with

$$
\begin{aligned}
& a(T-t)=\frac{2 k \bar{\lambda}}{\sigma^{2}} \ln \left(\frac{2 \gamma \mathrm{e}^{\frac{1}{2}(\gamma+k)(T-t)}}{(\gamma+k)\left(\mathrm{e}^{\gamma(T-t)}-1\right)+2 \gamma}\right) \\
& b(T-t)=\frac{2\left(\mathrm{e}^{\gamma(T-t)}-1\right)}{(\gamma+k)\left(\mathrm{e}^{\gamma(T-t)}-1\right)+2 \gamma}
\end{aligned}
$$

where $\gamma=\sqrt{k^{2}+2 \sigma^{2}}$.

\section{Pricing defaultable bonds (zero recovery)}
With stochastic intensities and interest rates it can be shown that with $R=0$

$$
\begin{aligned}
\tilde{B}_{t}^{T} & =\mathbb{E}_{t}\left[\mathrm{e}^{-\int_{t}^{T} r_{s} \mathrm{~d} s} \mathbb{1}_{\{\tau>T\}}\right] \\
& =\mathbb{E}_{t}\left[\mathbb{E}\left[\mathrm{e}^{-\int_{t}^{T} r_{s} \mathrm{~d} s} \mathbb{1}_{\{\tau>T\}} \mid \mathcal{G}_{T}\right]\right] \\
& =\mathbb{E}_{t}\left[\mathbb{E}\left[\mathrm{e}^{-\int_{t}^{T} r_{s} \mathrm{~d} s} \mathbb{1}_{\left\{N_{T}=0\right\}} \mid \mathcal{G}_{T}\right]\right] \\
& =\mathbb{E}_{t}\left[\mathbb{E}\left[\mathrm{e}^{-\int_{t}^{T} r_{s} \mathrm{~d} s} \mid \mathcal{G}_{T}\right] \mathbb{E}\left[\mathbb{1}_{\left\{N_{T}=0\right\}} \mid \mathcal{G}_{T}\right]\right] \\
& =\mathbb{E}_{t}\left[\mathbb{E}\left[\mathrm{e}^{-\int_{t}^{T} r_{s} \mathrm{~d} s} \mid \mathcal{G}_{T}\right] \mathrm{e}^{-\int_{t}^{T} \lambda_{s} \mathrm{~d} s}\right] \\
& =\mathbb{E}_{t}\left[\mathrm{e}^{-\int_{t}^{T} r_{s}+\lambda_{s} \mathrm{~d} s}\right]
\end{aligned}
$$

If the intensity process $\lambda_{t}$ and interest rate process $r_{t}$ are independent

$$
\tilde{B}_{t}^{T}=\mathbb{E}_{t}\left[\mathrm{e}^{-\int_{t}^{T} r_{s} \mathrm{~d} s}\right] \mathbb{E}_{t}\left[\mathrm{e}^{-\int_{t}^{T} \lambda_{s} \mathrm{~d} s}\right]=B_{t}^{T} \mathbb{Q}_{t}[\tau>T]
$$

as in the case with deterministic intensity.

\subsection{Additional stuff}
Possion process: Let $N_{t}$ be a poisson process with intensity parameter $\lambda$ (this means $N_{t} \sim p o(\lambda t)$ ). Then

$$
\mathbb{Q}\left[N_{t}=k\right]=\frac{(\lambda t)^{k}}{k!} \mathrm{e}^{-\lambda t}
$$
