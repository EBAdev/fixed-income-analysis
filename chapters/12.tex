\chapter{Credit derivatives}
The main idea when pricing a credit default swap (CDS) (a single name credit derivative) is that the value at time $t=0$ is zero. We set spreads so we are indifferent between being a protection seller or buyer (in expectation).

The protection buyer pays the CDS spread $s^{C D S}(T)$ every payment date unless the reference entity goes bankrupt. The value of the stream of cashflows is the sum of the discounted future payments times the probability of no default until time $t$. (Assuming the CDS does not carry any counterparty risk itself).

\subsection{Pricing CDS}
Let $\delta_{i}=T_{i}-T_{i-1}$. The protections buyer's cashflows are

$$
\begin{aligned}
\pi^{\text {prem }} & =\mathbb{E}^{\mathbb{Q}}\left[\sum_{i=1}^{N} \exp \left(-\int_{0}^{T_{i}} r_{s} \mathrm{~d} s\right) \mathbb{1}_{\left\{\tau>T_{i}\right\}} s^{C D S}\left(T_{N}\right) \delta_{i}\right] \\
& =s^{C D S}\left(T_{N}\right) \sum_{i=1}^{N} \delta_{i} \underbrace{\mathbb{E}^{\mathbb{Q}}\left[\exp \left(-\int_{0}^{T_{i}} r_{s}+\lambda_{s} \mathrm{~d} s\right)\right]}_{\text {price of defaultable ZCB, no recovery }} \\
& =s^{C D S}\left(T_{N}\right) \sum_{i=1}^{N} \delta_{i} B_{0}^{T_{i}} \mathbb{Q}_{0}\left[\tau>T_{i}\right]
\end{aligned}
$$

The protection seller pays $(1-R) \cdot$ Notional if there is a default. (We abstract from settlement, liquidity and other issues). Assuming recovery rate $R$ constant, the protection seller's cashflows are

$$
\begin{aligned}
\pi^{\text {prot }} & =\mathbb{E}^{\mathbb{Q}}\left[\exp \left(-\int_{0}^{\tau} r_{s} \mathrm{~d} s\right) \mathbb{1}_{\left\{\tau \leq T_{N}\right\}}(1-R)\right] \\
& =(1-R) \mathbb{E}^{\mathbb{Q}}\left[\int_{0}^{T_{N}} \lambda_{t} \exp \left(-\int_{0}^{t} r_{s}+\lambda_{s} \mathrm{~d} s\right) \mathrm{d} t\right]
\end{aligned}
$$

If $r$ and $\lambda$ are independent

$$
\pi^{\text {prot }}=(1-R) \int_{0}^{T_{N}} B_{o}^{t} \mathbb{Q}_{0}[\tau>t] \hat{\lambda}_{t} \mathrm{~d} t
$$

with $\hat{\lambda}(t)$ being the time-deterministic intensity (or expected intensity $\mathbb{E}_{0}^{\mathrm{Q}}\left[\lambda_{t}\right]$ ).\\
Equating the two $\pi^{\text {prem }}=\pi^{\text {prot }}$ we obtain

$$
\begin{aligned}
s^{C D S}\left(T_{N}\right) \sum_{i=1}^{N} \delta_{i} B_{0}^{T_{i}} \mathbb{Q}_{0}\left[\tau>T_{i}\right] & =(1-R) \int_{0}^{T_{N}} B_{o}^{t} \mathbb{Q}_{0}[\tau>t] \hat{\lambda}_{t} \mathrm{~d} t \\
& \mathbb{I} \\
s^{C D S}\left(T_{N}\right) & =\frac{(1-R) \int_{0}^{T_{N}} B_{o}^{t} \mathbb{Q}_{0}[\tau>t] \hat{\lambda}_{t} \mathrm{~d} t}{\sum_{i=1}^{N} \delta_{i} B_{0}^{T_{i}} \mathbb{Q}_{0}\left[\tau>T_{i}\right]}
\end{aligned}
$$

If $\lambda$ is constant $\mathbb{Q}_{0}\left[\tau>T_{i}\right]=\mathrm{e}^{-\lambda T_{i}}$. If $\lambda$ is affine

$$
\mathbb{Q}_{0}\left[\tau>T_{i}\right]=\mathbb{E}^{\mathbb{Q}}\left[\mathrm{e}^{-\int_{0}^{T_{i}} \lambda_{s} \mathrm{~d} s}\right]=\mathrm{e}^{A\left(0, T_{i}\right)+B\left(0, T_{i}\right) \lambda_{0}}
$$

\subsection{Marking to market}
To find the current value (time $t, t_{n-1} \leq t<t_{n}$ ) of the contract (an old one) from the perspective of the protection buyer, observe that

$$
V_{\text {old }}=V_{\text {old }}+V_{n e w},
$$

since the newly issued contract has zero market value. Thus

$$
\begin{aligned}
V_{\text {old }} & =\pi_{\text {old }}^{\text {prot }}-\pi_{\text {old }}^{\text {prem }}-\left(\pi_{\text {new }}^{\text {prot }}-\pi_{\text {new }}^{\text {prem }}\right) \\
& =\pi_{\text {new }}^{\text {prem }}-\pi_{\text {old }}^{\text {prem }} \\
& =\sum_{i=1}^{N} B_{t}^{T_{i}} \mathbb{Q}_{t}\left[\tau>T_{i}\right] \delta_{i} s_{t}^{C D S}\left(T_{N}\right)-\sum_{i=1}^{N} B_{t}^{T_{i}} \mathbb{Q}_{t}\left[\tau>T_{i}\right] \delta_{i} s_{0}^{C D S}\left(T_{N}\right) \\
& =\sum_{i=1}^{N} B_{t}^{T_{i}} \mathbb{Q}_{t}\left[\tau>T_{i}\right] \delta_{i}\left(s_{t}^{C D S}\left(T_{N}\right)-s_{0}^{C D S}\left(T_{N}\right)\right),
\end{aligned}
$$

since $\pi_{\text {old }}^{\text {prot }}=\pi_{\text {new }}^{\text {prot }}$. From this it can be seen that the value of the contract from the perspective of the protection buyer increases as the CDS spread increases.
